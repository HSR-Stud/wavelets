\section{Faltung}
Faltung:
\begin{equation}
  \nonumber
  y(t) = f(t) \ast g(t) = \int\limits_{-\infty}^{\infty}f(u) \cdot g(t-u) du
\end{equation}

Diskret:
\begin{equation}
  \nonumber
  y(i)=f(i) \ast g(i)=\sum\limits_{k=-\infty}^{\infty}f(k)\cdot g(i-k)
\end{equation}
