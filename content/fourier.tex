	e_k=\{\frac{1}{\sqrt{2 \pi}}, \frac{\cos(x)}{\sqrt{2 \pi}}, \frac{\sin(x)}{\sqrt{2 \pi}}, \frac{\cos(2x)}{\sqrt{2 \pi}}, \frac{\sin(2x)}{\sqrt{2 \pi}}, ... \} 
	\qquad \qquad
	f(x) = \sum_{k=0}^{\infty}\overbrace{\langle e_k|f \rangle}^{c_k} e_k
